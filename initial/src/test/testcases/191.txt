function isPrimeNumber(n:integer):boolean;
                        var i:integer;
                        begin
                            if (n<2) then return false;
                            for i:=2 to sqrt(n) do
                                if (n mod i = 0) then
                                    return false;
                            return true;
                        end
                    procedure main();
                        begin
                            print(isPrimeNumber(13));
                        end
